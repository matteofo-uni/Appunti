```swift
{
    Int r = 0;
    For i = 1 To n - 1
    {
        For j = i + 1 To n
        {
            For k = 1 To j
            {
                r++;
            }
        }
    }

    Return r;
}
```

# Programmazione ricorsiva
Calcoliamo l'n-esimo numero fattoriale ricorsivamente:
```swift
Int factorial(Int n)
{
    If (n == 1)
    {
        Return 1;
    }

    Return n * factorial(n);
}
```

Della sequenza di fibonacci:
```swift
Int fibonacci(Int n)
{
    If (n <= 1)
    {
        Return 1;
    }

    Return fib(n - 1) + fib(n - 2);
}
```
